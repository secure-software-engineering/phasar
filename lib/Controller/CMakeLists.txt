file(GLOB_RECURSE CONTROLLER_SRC *.h *.cpp)

  phasar_ifdside
  phasar_mono
  phasar_db
  phasar_pointer
  phasar_typehierarchy
  phasar_controlflow
  phasar_phasarllvm_utils
  phasar_utils
  phasar_analysis_strategy
)

  Core
  Support
)

  add_phasar_library(phasar_controller
    SHARED
    ${CONTROLLER_SRC}
  )

    STATIC
    ${CONTROLLER_SRC}
  )
endif()

  curl
  ${CMAKE_THREAD_LIBS_INIT}
  ${PHASAR_STD_FILESYSTEM}

  PROPERTIES
  LINKER_LANGUAGE CXX
